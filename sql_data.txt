CREATE DATABASE CORONA CHARACTER SET utf8 COLLATE utf8_general_ci;

USE CORONA;

CREATE TABLE ADMINISTRATOR(
ID VARCHAR(30) NOT NULL, 
PASSWORD VARCHAR(255) NOT NULL, 
ROLE ENUM('A') NOT NULL,
PRIMARY KEY(ID));


CREATE TABLE SUBMITTER(
ID VARCHAR(30) NOT NULL, 
PASSWORD VARCHAR(255) NOT NULL, 
NAME VARCHAR(30) NOT NULL, 
GENDER ENUM('M', 'F') NOT NULL, 
ADDRESS ENUM('강남', '서대문', '서초', '마포', '용산') NOT NULL, 
BIRTHDATE DATE NOT NULL, 
PNUM CHAR(11) NOT NULL, 
AGE INT NOT NULL, 
ROLE ENUM('R', 'H') NOT NULL,
RNAME VARCHAR(30),
RLOCATE ENUM('용산', '서대문', '강남'),
RPNUM VARCHAR(11),
ESCORE INT NOT NULL DEFAULT 0,
PRIMARY KEY(ID)); 



CREATE TABLE ESTIMATOR(
ID VARCHAR(30) NOT NULL, 
PASSWORD VARCHAR(255) NOT NULL, 
NAME VARCHAR(30) NOT NULL, 
GENDER ENUM('M', 'F') NOT NULL, 
ADDRESS ENUM('강남', '서대문', '서초', '마포', '용산') NOT NULL, 
BIRTHDATE DATE NOT NULL, 
PNUM CHAR(11) NOT NULL, 
AGE INT NOT NULL, 
ROLE ENUM('E') NOT NULL,
PRIMARY KEY(ID));



CREATE TABLE TASKSCHEMA(
ROLE ENUM('R', 'H') NOT NULL,
SCHEMA1 VARCHAR(20),
SCHEMA2 VARCHAR(20),
SCHEMA3 VARCHAR(20), 
SCHEMA4 VARCHAR(20),
PRIMARY KEY(ROLE)); 



CREATE TABLE TASK( 
NAME VARCHAR(50) NOT NULL,
DESCRIPTION VARCHAR(100) NOT NULL,
TASKDATATABLE VARCHAR(30) NOT NULL,
CYCLE TIME NOT NULL,
ROLE ENUM('R', 'H') NOT NULL, 
PRIMARY KEY(NAME), 
FOREIGN KEY(ROLE) REFERENCES TASKSCHEMA(ROLE));



CREATE TABLE REQUEST(
TASKNAME VARCHAR(50) NOT NULL,
SUBMITTERID VARCHAR(30) NOT NULL,
ROLE ENUM('R', 'H') NOT NULL,
PRIMARY KEY(TASKNAME, SUBMITTERID), 
FOREIGN KEY(TASKNAME) REFERENCES TASK(NAME),  
FOREIGN KEY(SUBMITTERID) REFERENCES SUBMITTER(ID) ON DELETE CASCADE);



CREATE TABLE APPROVAL(
TASKNAME VARCHAR(50) NOT NULL,
SUBMITTERID VARCHAR(30) NOT NULL,
ROLE ENUM('R', 'H') NOT NULL,
PRIMARY KEY(TASKNAME, SUBMITTERID), 
FOREIGN KEY(TASKNAME) REFERENCES TASK(NAME),  
FOREIGN KEY(SUBMITTERID) REFERENCES SUBMITTER(ID) ON DELETE CASCADE);


CREATE TABLE TYPEORIGIN(
ID INT NOT NULL AUTO_INCREMENT,
SCHEMA1 VARCHAR(30),
SCHEMA2 VARCHAR(30),
SCHEMA3 VARCHAR(30),
SCHEMA4 VARCHAR(30),
SCHEMA5 VARCHAR(30),
SCHEMA6 VARCHAR(30),
SCHEMA7 VARCHAR(30),
SCHEMA8 VARCHAR(30),
SCHEMA9 VARCHAR(30),
SCHEMA10 VARCHAR(30),
CPNUM INT NOT NULL,
CADDRESS INT NOT NULL,
CVDATE INT NOT NULL,
CEXTRA INT NOT NULL,
SCHEMANUM INT NOT NULL,
PRIMARY KEY(ID));



CREATE TABLE RPARSING(
PARSINGID INT NOT NULL,
TASKNAME VARCHAR(50) NOT NULL, 
SID VARCHAR(30) NOT NULL,
TID INT NOT NULL, 
ROUND INT NOT NULL,
TERMSTART DATE NOT NULL,
TERMEND DATE NOT NULL,
EID VARCHAR(30) NOT NULL, 
TOTALTUPLE INT NOT NULL, 
DUPTUPLE INT NOT NULL, 
NULLRATE DOUBLE NOT NULL, 
ESTATE ENUM('Y', 'N') NOT NULL, 
ESCORE INT NOT NULL DEFAULT 0, 
PASS ENUM('P', 'N') NOT NULL DEFAULT 'N', 
PRIMARY KEY(PARSINGID),
FOREIGN KEY(TASKNAME) REFERENCES TASK(NAME), 
FOREIGN KEY(SID) REFERENCES SUBMITTER(ID) ON DELETE CASCADE, 
FOREIGN KEY(EID) REFERENCES ESTIMATOR(ID) ON DELETE CASCADE);



CREATE TABLE HPARSING(
PARSINGID INT NOT NULL,
TASKNAME VARCHAR(50) NOT NULL, 
SID VARCHAR(30) NOT NULL,
TID INT NOT NULL, 
ROUND INT 	NOT NULL,
TERMSTART DATE NOT NULL,
TERMEND DATE NOT NULL,
EID VARCHAR(30) NOT NULL, 
TOTALTUPLE INT NOT NULL, 
DUPTUPLE INT NOT NULL, 
NULLRATE DOUBLE NOT NULL, 
ESTATE ENUM('Y', 'N') NOT NULL, 
ESCORE INT NOT NULL DEFAULT 0, 
PASS ENUM('P', 'N') NOT NULL DEFAULT 'N', 
PRIMARY KEY(PARSINGID),
FOREIGN KEY(TASKNAME) REFERENCES TASK(NAME), 
FOREIGN KEY(SID) REFERENCES SUBMITTER(ID) ON DELETE CASCADE, 
FOREIGN KEY(EID) REFERENCES ESTIMATOR(ID) ON DELETE CASCADE);
